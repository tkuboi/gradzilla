public enum Status {
    QUEUED("queued"),
    SKIPPED("skipped"),
    DOWNLOADING("downloading"),
    DOWNLOADED("downloaded"),
    DOWNLOAD_FAILED("download failed"),
    CUTTING("cutting"),
    CUT("cut"),
    CUTTING_FAILED("cutting failed"),
    TASKING("tasking"),
    TASKING_FAILED("tasking failed"),
    TASKED("tasked"),
    DELETED("deletd"),
    DIARIZED("diarized"),
    ARCHIVED("archived"),
    APPROVED("approved"),
    TRANSCRIPTION_FAILED("transcription failed"),
    REQUEST_INCOMPLETE("request incomplete");

    private String text;

    Status(String text) {
      this.text = text;
    }

    public String toString() {
      return text;
    }
  }

private static void flow(InputStream is, OutputStream os, byte[] buf) throws IOException {
    int numRead;
    while ( (numRead = is.read(buf) ) >= 0) {
      os.write(buf, 0, numRead);
    }
  }
